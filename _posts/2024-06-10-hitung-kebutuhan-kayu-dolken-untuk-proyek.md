formula_dasar:
  konsep_penting:
  - label: Jumlah Batang
    formula: Total Panjang Needed ÷ Panjang per Batang
  - label: Buffer
    formula: 10-15% dari total (untuk cutting loss, kerusakan, dll)
  - label: Final Order
    formula: Jumlah Batang + Buffer
  standar_panjang: 4 meter per batang
perhitungan_aplikasi:
- nomor: 1
  nama: Pagar Vertikal
  icon: bi-bounding-box
  warna: primary
  scenario: Pagar taman setinggi 1.5 meter, panjang 10 meter
  steps:
  - step: 'Step 1: Hitung jumlah dolken per meter'
    perhitungan:
    - Diameter dolken = 6 cm = 0.06 meter
    - Spacing yang diinginkan = 5 cm = 0.05 meter
    - Pitch (dolken + spacing) = 0.06 + 0.05 = 0.11 meter
    - Jumlah per meter = 1 ÷ 0.11 = 9.09 ≈ 9 batang per meter
  - step: 'Step 2: Total untuk panjang pagar'
    perhitungan:
    - Total per row = 9 batang × 10 meter = 90 batang
  - step: 'Step 3: Adjust untuk tinggi'
    perhitungan:
    - Tinggi pagar = 1.5 meter
    - Panjang dolken = 4 meter
    - 1 batang bisa untuk 4 ÷ 1.5 = 2.67 ≈ 2 pieces (dengan sisa)
    - Actual kebutuhan = 90 ÷ 2 = 45 batang utuh
  - step: 'Step 4: Tambahkan buffer 15%'
    perhitungan:
    - Buffer = 45 × 15% = 6.75 ≈ 7 batang
    - Total order = 45 + 7 = 52 batang
  final: Order 52 batang dolken 6-8 cm
  budget: 52 × Rp 30.000 = Rp 1.560.000
- nomor: 2
  nama: Feature Wall Dekoratif
  icon: bi-grid-3x3-gap-fill
  warna: success
  scenario: Wall cladding 3m × 4m = 12 m²
  instalasi_type: Vertical Installation
  steps:
  - step: 'Step 1: Hitung coverage per batang'
    perhitungan:
    - Diameter = 3 cm = 0.03 m
    - Panjang = 4 m
    - Coverage per batang = 0.03 × 4 = 0.12 m²
  - step: 'Step 2: Total batang needed'
    perhitungan:
    - Luas wall = 12 m²
    - Batang needed = 12 ÷ 0.12 = 100 batang
  - step: 'Step 3: Adjust untuk spacing (jika ada)'
    keterangan: 'Jika pakai spacing 1 cm antara dolken:'
    perhitungan:
    - Effective coverage = (0.03 + 0.01) × 4 = 0.16 m²
    - Batang needed = 12 ÷ 0.16 = 75 batang
  - step: 'Step 4: Buffer'
    perhitungan:
    - Buffer 10% = 75 × 10% = 7.5 ≈ 8 batang
    - Total = 75 + 8 = 83 batang
  final: Order 85 batang dolken 2-3 cm (rounded up)
  budget: 85 × Rp 15.000 = Rp 1.275.000
- nomor: 3
  nama: Gazebo
  icon: bi-house-fill
  warna: info
  scenario: Gazebo 3m × 3m, tinggi 3m
  breakdown: Breakdown Components
  komponen:
  - nama: Tiang Utama (8-10 cm)
    perhitungan:
    - 4 tiang utama @ 3 meter tinggi
    - Kebutuhan = 4 × 3m = 12 meter linear
    - Batang needed = 12 ÷ 4 = 3 batang
    order: 4 batang (include 1 buffer)
    cost: 4 × Rp 35.000 = Rp 140.000
  - nama: Rangka Atap (6-8 cm)
    perhitungan:
    - Perimeter = (3 + 3 + 3 + 3) = 12 meter
    - Cross beams = 2 × 3 meter = 6 meter
    - Diagonal supports = 2 × 4.24 meter = 8.48 meter
    - Total = 12 + 6 + 8.48 = 26.48 meter
    - Batang = 26.48 ÷ 4 = 6.62 ≈ 7 batang
    - Buffer 15% = 7 × 1.15 = 8.05 ≈ 9 batang
    order: 9 batang
    cost: 9 × Rp 30.000 = Rp 270.000
  - nama: Dekorasi Railing (4-6 cm)
    perhitungan:
    - Perimeter = 12 meter
    - Vertical slats @ 10cm spacing = 120 slats
    - Height per slat = 1 meter
    - Total linear = 120 × 1 = 120 meter
    - Batang = 120 ÷ 4 = 30 batang
    - Buffer 10% = 30 × 1.1 = 33 batang
    order: 33 batang
    cost: 33 × Rp 25.000 = Rp 825.000
  total_material: Rp 1.235.000
  total_batang: 46 batang (mixed sizes)
- nomor: 4
  nama: Pergola
  icon: bi-columns-gap
  warna: warning
  scenario: Pergola 4m × 3m
  komponen:
  - nama: Tiang (10-12 cm)
    perhitungan:
    - 4 tiang @ 3m = 12 meter linear
    - Batang = 12 ÷ 4 = 3 batang
    order: 4 batang (include buffer)
    cost: 4 × Rp 45.000 = Rp 180.000
  - nama: Beam Utama (8-10 cm)
    perhitungan:
    - 2 beams @ 4m = 8 meter
    - Batang = 8 ÷ 4 = 2 batang
    order: 3 batang (include buffer)
    cost: 3 × Rp 35.000 = Rp 105.000
  - nama: Rafters (6-8 cm)
    perhitungan:
    - Spacing 40cm = 4 ÷ 0.4 = 10 rafters
    - "@ 3m each = 30 meter linear"
    - Batang = 30 ÷ 4 = 7.5 ≈ 8 batang
    order: 9 batang (with buffer)
    cost: 9 × Rp 30.000 = Rp 270.000
  total_material: Rp 555.000
  total_batang: 16 batang
- nomor: 5
  nama: Ceiling Treatment
  icon: bi-grid-fill
  warna: purple
  scenario: Cafe ceiling 8m × 6m = 48 m², parallel pattern
  steps:
  - step: 'Step 1: Determine spacing'
    perhitungan:
    - Spacing between dolken = 20 cm
  - step: 'Step 2: Calculate rows needed'
    perhitungan:
    - Width = 6 meter = 600 cm
    - Rows = 600 ÷ 20 = 30 rows
  - step: 'Step 3: Length per row'
    perhitungan:
    - Length = 8 meter
  - step: 'Step 4: Total batang'
    perhitungan:
    - Total linear = 30 rows × 8m = 240 meter
    - Batang (4m each) = 240 ÷ 4 = 60 batang
    - Buffer 10% = 60 × 1.1 = 66 batang
  final: Order 66 batang dolken 3-4 cm
  budget: 66 × Rp 25.000 = Rp 1.650.000
calculator_tool:
  formulas:
  - jenis: Pagar Horizontal/Vertikal
    formula: Batang = (Panjang × Tinggi) ÷ (Diameter + Spacing) ÷ Panjang_Batang ×
      Buffer
  - jenis: Wall Cladding
    formula: Batang = Luas_Wall ÷ (Diameter × Panjang_Batang) × Buffer
  - jenis: Ceiling
    formula: Batang = (Total_Linear_Length ÷ Panjang_Batang) × Buffer
  - jenis: Struktur (Gazebo/Pergola)
    formula: Breakdown per komponen, sum total

common_mistakes:
- nomor: 1
  mistake: Lupa Hitung Cutting Loss
  icon: bi-scissors
  warna: danger
  issue: Assume 100% material usable
  reality: Selalu ada waste dari cutting
  solution: Add 10-15% buffer
- nomor: 2
  mistake: Tidak Pertimbangkan Spacing
  icon: bi-distribute-horizontal
  warna: warning
  issue: Hitung rapat-rapat tanpa gap
  reality: Design biasanya butuh spacing untuk estetika/sirkulasi
  solution: Include spacing in calculation
- nomor: 3
  mistake: Salah Unit
  icon: bi-calculator
  warna: info
  issue: Mix meter, cm, inch
  reality: Chaos!
  solution: 'Standardize (recommend: meter untuk all)'
- nomor: 4
  mistake: Tidak Pertimbangkan Struktur Pendukung
  icon: bi-box-seam
  warna: primary
  issue: Hanya hitung material utama
  reality: Butuh crossbeams, supports, dll
  solution: List all components
