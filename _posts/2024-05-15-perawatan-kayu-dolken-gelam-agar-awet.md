perawatan_aplikasi:
  outdoor:
    judul: Aplikasi Outdoor
    subtitle: Perawatan untuk dolken yang terpapar cuaca
    warna: primary
    treatments:
    - nama: Pembersihan Rutin
      frekuensi: Setiap 1-2 bulan
      icon: bi-droplet-fill
      steps:
      - Sikat dengan sikat berbulu lembut untuk hilangkan debu
      - Cuci dengan air bersih menggunakan selang
      - Untuk kotoran membandel, gunakan sabun kayu mild
      - Bilas hingga bersih
      - Biarkan kering natural (jangan di bawah matahari langsung)
      dont:
      - Gunakan pressure washer (bisa merusak serat)
      - Pakai chemical harsh (bleach, dll)
      - Sikat dengan wire brush
    - nama: Aplikasi Minyak Kayu
      frekuensi: Setiap 6-12 bulan
      icon: bi-droplet-half
      fungsi:
      - Melindungi dari kerusakan sinar UV
      - Mempertahankan warna alami
      - Mencegah kayu terlalu kering
      - Memberi tampilan satin yang indah
      produk:
      - Minyak Jati (Teak Oil)
      - Minyak Biji Rami (Linseed Oil)
      - Minyak Pelindung Kayu khusus outdoor
      steps:
      - Bersihkan kayu hingga kering sempurna
      - Amplas ringan (grit 220) jika permukaan kasar
      - Aplikasikan minyak dengan kuas atau kain
      - Biarkan meresap 15-30 menit
      - Lap kelebihan minyak yang tidak meresap
      - Biarkan kering 24 jam sebelum terpapar air
      - Aplikasi kedua (opsional) setelah 48 jam
      tips:
      - Lakukan di pagi atau sore (hindari terik matahari)
      - Gunakan sarung tangan
      - Aplikasi lebih tipis 2x lebih baik dari 1x tebal
    - nama: Inspeksi Berkala
      frekuensi: Setiap 6 bulan
      icon: bi-clipboard-check-fill
      checkpoints:
      - judul: Retak atau Belah
        items:
        - Periksa permukaan untuk retakan
        - Retakan kecil adalah normal (pergerakan kayu)
        - Retakan besar perlu pengisi atau penggantian
      - judul: Sambungan Longgar
        items:
        - Cek sambungan, baut, atau pengikat
        - Kencangkan jika ada yang longgar
        - Ganti perangkat keras yang berkarat
      - judul: Warna & Tekstur
        items:
        - Perubahan warna drastis = terlalu terpapar
        - Permukaan berbulu = perlu amplas & aplikasi ulang minyak
        - Bercak hitam = jamur/lumut (jarang di gelam, tapi mungkin terjadi)
      - judul: Integritas Struktural
        items:
        - Uji kekuatan dengan tekanan lembut
        - Perhatikan lengkungan atau pembelokan
        - Pastikan tidak ada bagian yang lapuk
  indoor:
    judul: Aplikasi Indoor
    subtitle: Perawatan untuk dolken di dalam ruangan
    warna: success
    treatments:
    - nama: Pembersihan Rutin
      frekuensi: Setiap minggu
      icon: bi-stars
      steps:
      - Lap dengan kain mikrofiber kering
      - Untuk kotoran, gunakan kain lembab (sedikit basah)
      - Keringkan dengan kain kering
      - Tidak perlu pembersih kimia
      monthly:
      - Vakum dengan attachment sikat lembut
      - Lap dengan pembersih kayu ringan
      - Keringkan dengan sempurna
      - Aplikasikan semir furniture (opsional)
    - nama: Perawatan Kondisi
      frekuensi: Setiap 3-6 bulan
      icon: bi-heart-pulse-fill
      wall_ceiling:
      - Bersihkan debu dengan mikrofiber
      - Semprotkan ringan kondisioner kayu
      - Gosok dengan kain lembut
      furniture:
      - Bersihkan dengan menyeluruh
      - Aplikasikan kondisioner kayu atau pasta lilin
      - Gosok hingga mengkilap
      - Melindungi dan mempertahankan kilau
perawatan_usia:
- periode: Tahun 1-5
  subtitle: Masa Settling
  warna: info
  icon: bi-hourglass-split
  karakteristik:
  - Kayu masih 'bergerak' menyesuaikan lingkungan
  - Mungkin ada sedikit pembelokan atau puntiran
  - Warna mulai mengembangkan karakter
  perawatan:
  - Pemantauan ekstra untuk pergerakan
  - Perawatan minyak lebih sering (setiap 6 bulan) untuk outdoor
  - Kencangkan ulang perangkat keras jika ada yang longgar
  - Dokumentasikan kondisi awal untuk referensi masa depan
- periode: Tahun 5-10
  subtitle: Stabilisasi Selesai
  warna: primary
  icon: bi-check-circle-fill
  karakteristik:
  - Kayu sudah sepenuhnya stabil
  - Pergerakan minimal, struktur terkunci
  - Warna sudah berkembang sempurna
  - Siap untuk performa optimal
  perawatan:
  - Transisi ke jadwal perawatan standar
  - Perawatan minyak bisa dikurangi ke 8-12 bulan
  - Inspeksi rutin setiap 6 bulan
  - Dokumentasi kondisi optimal sebagai acuan
- periode: Tahun 10-20
  subtitle: Prime Time
  warna: success
  icon: bi-trophy-fill
  karakteristik:
  - Performa optimal dan konsisten
  - Warna kaya dan matang
  - Tanpa masalah jika perawatan tepat
  perawatan:
  - Pertahankan rutinitas pembersihan & peminyakan
  - Jadwal inspeksi standar
  - Nikmati keindahannya!
- periode: Tahun 20+
  subtitle: Mature Wood
  warna: warning
  icon: bi-gem
  karakteristik:
  - Warna kaya dan dalam dengan karakter
  - Mungkin ada pelapukan alami (menambah pesona)
  - Tetap kuat secara struktural
  - Ketahanan terbukti
  perawatan:
  - Inspeksi lebih sering (setiap 3-4 bulan)
  - Pengamplasan ringan untuk peremajaan
  - Pewarnaan ulang (opsional)
  - Peminyakan intensif
  - Pertimbangkan memperbarui perangkat keras
  - Ganti bagian individual jika ada retakan signifikan

  - Pertimbangkan aplikasi perawatan anti air
- musim: Musim Kemarau
  icon: bi-sun-fill
  warna: warning
  challenges:
  - Paparan sinar UV tinggi
  - Kayu bisa terlalu kering
  - Akumulasi debu
  extra_care:
  - Perawatan minyak lebih sering
  - Kain peneduh untuk area matahari ekstrem (opsional)
  - Pembersihan debu lebih sering
  - Pantau retakan akibat kekeringan

common_mistakes:
- nomor: 1
  mistake: Perawatan Berlebihan
  icon: bi-exclamation-triangle-fill
  warna: warning
  detail: Aplikasi minyak terlalu sering atau terlalu tebal
  akibat:
  - Penumpukan yang lengket
  - Menarik debu
  - Tampilan tidak alami
  solution: Patuhi jadwal yang direkomendasikan, lebih sedikit lebih baik
- nomor: 2
  mistake: Mengabaikan Drainase
  icon: bi-droplet-fill
  warna: primary
  detail: Membiarkan air menggenang di sekitar kayu
  akibat:
  - Meski tahan air, paparan berkepanjangan tetap tidak baik
  - Bisa menyebabkan pembelokan
  - Menciptakan lingkungan untuk jamur
  solution: Pastikan drainase tepat, tinggikan kayu dari tanah jika perlu
- nomor: 3
  mistake: Metode Pembersihan Kasar
  icon: bi-x-circle-fill
  warna: danger
  detail: Pencuci tekanan tinggi, sikat kawat, bahan kimia keras
  akibat:
  - Merusak serat kayu
  - Menghilangkan minyak alami
  - Memperpendek umur pakai
  solution: Metode lembut, bersabar
- nomor: 4
  mistake: Perawatan Tidak Konsisten
  icon: bi-calendar-x-fill
  warna: info
  detail: Rajin 1 tahun, kemudian mengabaikan
  akibat:
  - Penuaan tidak merata
  - Kegagalan mendadak
  - Perbaikan mahal
  solution: Atur pengingat kalender, jadikan rutinitas
